pod power = power->new();

# Switch power profiles on battery and AC
# - also disable/re-enable sync thing since it uses a decent amount of battery
power->on_battery => power->change_power_profile(profile: "power-saver");
power->on_battery => bash {
    systemctl --user stop syncthing
};

power->on_ac => power->change_power_profile(profile: "balanced");
power->on_ac => bash {
    # force syncthing to give up its port bindings if there is still anything
    while pgrep -x syncthing >/dev/null; do sleep 2; done
    systemctl --user restart syncthing
};
